# DoH Proxy Integration — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** When OKX API domains are blocked (DNS pollution or TLS RST), onchainos CLI transparently routes HTTP and WebSocket traffic through DoH-resolved proxy nodes.

**Architecture:** New `src/doh/` module with 6 files (types, binary, cache, manager, ws, mod). DohManager reads cache → tries direct → on failure calls `okx-doh-resolver` binary → caches proxy node → rewrites requests. ApiClient, WalletApiClient, and WS daemon each integrate DohManager independently.

**Tech Stack:** Rust, reqwest 0.12 (existing), tokio-rustls 0.26 (new direct dep, already transitive), rustls 0.23 (already transitive), webpki-roots (already transitive)

**Spec:** `docs/superpowers/specs/2026-04-14-doh-proxy-design.md`

```rust
pub struct DohNode {

    Direct,
```

```rust
use crate::home::onchainos_home;
use super::types::{DohCacheEntry, DohCacheFile, DohMode, DohNode, FailedNode};

const CACHE_FILENAME: &str = "doh-cache.json";
const FAILED_NODE_TTL_MS: u64 = 3_600_000; // 1 hour

fn cache_path() -> Option<std::path::PathBuf> {
    onchainos_home().ok().map(|h| h.join(CACHE_FILENAME))
}

fn now_ms() -> u64 {
    std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_millis() as u64
}

/// Read cache entry for a domain. Cleans up expired failed nodes.
/// Returns None on any error (missing file, parse error, missing domain).
pub fn read_cache(domain: &str) -> Option<DohCacheEntry> {
    let path = cache_path()?;
    let data = std::fs::read_to_string(&path).ok()?;
    let mut file: DohCacheFile = serde_json::from_str(&data).ok()?;
    let entry = file.get_mut(domain)?;
    // Clean expired failed nodes
    let now = now_ms();
    entry.failed_nodes.retain(|n| now - n.failed_at < FAILED_NODE_TTL_MS);
    Some(entry.clone())
}

/// Write/merge a cache entry for a domain. Atomic: write .tmp then rename.
/// Best-effort: silently ignores all errors.
pub fn write_cache(domain: &str, entry: &DohCacheEntry) {
    let Some(path) = cache_path() else { return };

    // Read existing, merge
    let mut file: DohCacheFile = path
        .exists()
        .then(|| std::fs::read_to_string(&path).ok())
        .flatten()
        .and_then(|d| serde_json::from_str(&d).ok())
        .unwrap_or_default();

    file.insert(domain.to_string(), entry.clone());

    let Ok(json) = serde_json::to_string_pretty(&file) else { return };

    let tmp = path.with_extension("json.tmp");
    if std::fs::write(&tmp, json.as_bytes()).is_ok() {
        let _ = std::fs::rename(&tmp, &path);
    }
}

/// Remove a domain's cache entry.
pub fn invalidate_cache(domain: &str) {
    let Some(path) = cache_path() else { return };
    let Ok(data) = std::fs::read_to_string(&path) else { return };
    let Ok(mut file) = serde_json::from_str::<DohCacheFile>(&data) else { return };
    if file.remove(domain).is_some() {
        if let Ok(json) = serde_json::to_string_pretty(&file) {
            let tmp = path.with_extension("json.tmp");
            if std::fs::write(&tmp, json.as_bytes()).is_ok() {
                let _ = std::fs::rename(&tmp, &path);
            }
        }
    }
}
```

```rust
use anyhow::Result;
use super::types::{DohBinaryResponse, DohNode};
use crate::home::onchainos_home;
use std::path::PathBuf;
use std::time::Duration;

const BINARY_NAME: &str = "okx-doh-resolver";

const CDN_SOURCES: &[&str] = &[
    "https://static.okx.com/upgradeapp/doh",
    "https://pcdoh.qcxex.com/upgradeapp/doh",
    "https://static.coinall.ltd/upgradeapp/doh",
];

/// Resolve the binary path. OKX_DOH_BINARY_PATH env var overrides default.
pub fn binary_path() -> Option<PathBuf> {
    if let Ok(p) = std::env::var("OKX_DOH_BINARY_PATH") {
        return Some(PathBuf::from(p));
    }
    onchainos_home().ok().map(|h| h.join("bin").join(BINARY_NAME))
}

/// Map Rust compile target to CDN platform string.
fn cdn_platform() -> Option<&'static str> {
    #[cfg(all(target_os = "macos", target_arch = "aarch64"))]
    { return Some("darwin-arm64"); }
    #[cfg(all(target_os = "macos", target_arch = "x86_64"))]
    { return Some("darwin-x64"); }
    #[cfg(all(target_os = "linux", target_arch = "x86_64"))]
    { return Some("linux-x64"); }
    #[cfg(all(target_os = "windows", target_arch = "x86_64"))]
    { return Some("win32-x64"); }
    #[allow(unreachable_code)]
    None
}

/// Download the binary from CDN. Tries each source in order.
/// Best-effort: returns Ok(()) on success, Err on all-failed.
pub async fn download_binary() -> Result<()> {
    let dest = binary_path().ok_or_else(|| anyhow::anyhow!("cannot determine binary path"))?;
    let platform = cdn_platform().ok_or_else(|| anyhow::anyhow!("unsupported platform"))?;

    // Ensure parent dir exists
    if let Some(parent) = dest.parent() {
        std::fs::create_dir_all(parent)?;
    }

    let client = reqwest::Client::builder()
        .timeout(Duration::from_secs(30))
        .build()?;

    for cdn in CDN_SOURCES {
        let url = format!("{}/{}/{}", cdn, platform, BINARY_NAME);
        eprintln!("[doh] downloading {} ...", url);
        match client.get(&url).send().await {
            Ok(resp) if resp.status().is_success() => {
                let bytes = resp.bytes().await?;
                let tmp = dest.with_extension("tmp");
                std::fs::write(&tmp, &bytes)?;
                #[cfg(unix)]
                {
                    use std::os::unix::fs::PermissionsExt;
                    std::fs::set_permissions(&tmp, std::fs::Permissions::from_mode(0o755))?;
                }
                std::fs::rename(&tmp, &dest)?;
                eprintln!("[doh] binary saved to {}", dest.display());
                return Ok(());
            }
            Ok(resp) => {
                eprintln!("[doh] {} returned HTTP {}", url, resp.status());
            }
            Err(e) => {
                eprintln!("[doh] {} failed: {}", url, e);
            }
        }
    }
    anyhow::bail!("all CDN sources failed")
}

/// Execute the binary and return the resolved node.
/// Returns None on any error (binary missing, timeout, bad output, non-zero code).
pub async fn exec_doh_binary(
    domain: &str,
    exclude: &[String],
) -> Option<DohNode> {
    let path = binary_path()?;
    if !path.exists() {
        return None;
    }

    let mut cmd = std::process::Command::new(&path);
    cmd.arg("--domain").arg(domain);
    if !exclude.is_empty() {
        cmd.arg("--exclude").arg(exclude.join(","));
    }

    // Run with timeout via tokio::task::spawn_blocking
    let output = tokio::time::timeout(
        Duration::from_secs(30),
        tokio::task::spawn_blocking(move || cmd.output()),
    )
    .await
    .ok()? // timeout
    .ok()? // join error
    .ok()?; // io error

    if !output.status.success() {
        return None;
    }

    let resp: DohBinaryResponse = serde_json::from_slice(&output.stdout).ok()?;
    if resp.code != 0 || resp.data.ip.is_empty() {
        return None;
    }

    Some(DohNode {
        ip: resp.data.ip,
        host: resp.data.host,
        ttl: resp.data.ttl,
    })
}
```

```rust
use std::net::SocketAddr;
use super::types::{DohCacheEntry, DohMode, DohNode, FailedNode};
use super::{binary, cache};

/// Manages DoH proxy state for a single domain.
pub struct DohManager {
    domain: String,
    original_base_url: String,
    mode: Option<DohMode>,
    node: Option<DohNode>,
    resolved: bool,
    retried: bool,
    /// Whether user configured a custom base_url (skips DoH entirely).
    custom_base_url: bool,
}

fn now_ms() -> u64 {
    std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_millis() as u64
}

impl DohManager {
    /// Create a new DohManager.
    /// `base_url` is the original URL (e.g. "https://web3.okx.com").
    /// `custom_base_url` should be true if the user explicitly set OKX_BASE_URL or --base-url.
    pub fn new(domain: &str, base_url: &str, custom_base_url: bool) -> Self {
        Self {
            domain: domain.to_string(),
            original_base_url: base_url.to_string(),
            mode: None,
            node: None,
            resolved: false,
            retried: false,
            custom_base_url,
        }
    }

    /// Called before first request. Reads cache, sets mode/node.
    /// If custom base_url is set, does nothing (DoH skipped).
    pub fn prepare(&mut self) {
        if self.custom_base_url || self.resolved {
            return;
        }
        self.resolved = true;

        if let Some(entry) = cache::read_cache(&self.domain) {
            self.mode = Some(entry.mode.clone());
            self.node = entry.node;
        }
        // If no cache: mode stays None, will try direct first.
    }

    /// Called on network failure. Resolves DoH and returns whether caller should retry.
    ///
    /// Flow:
    /// 1. If custom base_url or already retried this cycle → false
    /// 2. If current node exists → add to failed, re-resolve excluding failed
    /// 3. If no node (first failure) → call binary
    /// 4. Binary returns node → cache it, set mode=Proxy, reset retried → true
    /// 5. Binary returns None → fallback direct → false
    pub async fn handle_failure(&mut self) -> bool {
        if self.custom_base_url || self.retried {
            return false;
        }

        // Collect failed node IPs for --exclude
        let mut exclude: Vec<String> = vec![];
        if let Some(cache_entry) = cache::read_cache(&self.domain) {
            exclude.extend(cache_entry.failed_nodes.iter().map(|n| n.ip.clone()));
        }

        // If current node failed, add it to failed list
        if let Some(ref current) = self.node {
            let ip = current.ip.clone();
            if !exclude.contains(&ip) {
                exclude.push(ip.clone());
            }
            // Persist the failure
            let mut entry = cache::read_cache(&self.domain).unwrap_or(DohCacheEntry {
                mode: DohMode::Proxy,
                node: self.node.clone(),
                failed_nodes: vec![],
                updated_at: now_ms(),
            });
            if !entry.failed_nodes.iter().any(|n| n.ip == ip) {
                entry.failed_nodes.push(FailedNode {
                    ip,
                    failed_at: now_ms(),
                });
            }
            cache::write_cache(&self.domain, &entry);
        }

        // Ensure binary exists, download if needed
        if binary::binary_path().map_or(true, |p| !p.exists()) {
            if binary::download_binary().await.is_err() {
                // Can't get binary → give up
                return false;
            }
        }

        // Call binary
        match binary::exec_doh_binary(&self.domain, &exclude).await {
            Some(new_node) => {
                let entry = DohCacheEntry {
                    mode: DohMode::Proxy,
                    node: Some(new_node.clone()),
                    failed_nodes: exclude
                        .iter()
                        .map(|ip| FailedNode {
                            ip: ip.clone(),
                            failed_at: now_ms(),
                        })
                        .collect(),
                    updated_at: now_ms(),
                };
                cache::write_cache(&self.domain, &entry);
                self.mode = Some(DohMode::Proxy);
                self.node = Some(new_node);
                self.retried = true;
                true
            }
            None => {
                // All nodes exhausted or binary failed — fallback to direct
                // Don't write cache (avoid dead failedNodes loop per spec)
                self.retried = true;
                false
            }
        }
    }

    /// Returns the base_url to use for requests.
    /// - Proxy mode: https://{node.host}
    /// - Direct/None: original base_url
    pub fn base_url(&self) -> &str {
        if self.mode == Some(DohMode::Proxy) {
            if let Some(ref node) = self.node {
                // Caller should use this with resolve_override()
                // Return base_url rewritten to proxy host
                // We can't return a &str to a dynamically built String,
                // so we store it. See proxy_base_url().
                return self.original_base_url.as_str();
            }
        }
        &self.original_base_url
    }

    /// Returns the proxy base_url as an owned String when in proxy mode.
    pub fn proxy_base_url(&self) -> Option<String> {
        if self.mode == Some(DohMode::Proxy) {
            if let Some(ref node) = self.node {
                return Some(format!("https://{}", node.host));
            }
        }
        None
    }

    /// Returns resolve override for reqwest ClientBuilder.
    /// When in proxy mode, maps node.host → node.ip:443.
    pub fn resolve_override(&self) -> Option<(String, SocketAddr)> {
        if self.mode == Some(DohMode::Proxy) {
            if let Some(ref node) = self.node {
                if let Ok(ip) = node.ip.parse() {
                    return Some((node.host.clone(), SocketAddr::new(ip, 443)));
                }
            }
        }
        None
    }

    /// Call after a successful direct connection when there's no cache yet.
    /// Caches mode=Direct so future requests skip DoH.
    pub fn cache_direct_if_needed(&self) {
        if self.custom_base_url {
            return;
        }
        // Only cache direct if we have no mode set (first successful direct)
        if self.mode.is_none() {
            let entry = DohCacheEntry {
                mode: DohMode::Direct,
                node: None,
                failed_nodes: vec![],
                updated_at: now_ms(),
            };
            cache::write_cache(&self.domain, &entry);
        }
    }

    /// Reset the retried flag. Called after successful proxy switch
    /// to support long-running MCP processes that may need to failover again later.
    pub fn reset_retried(&mut self) {
        self.retried = false;
    }

    /// Whether DoH is active (has a proxy node).
    pub fn is_proxy(&self) -> bool {
        self.mode == Some(DohMode::Proxy) && self.node.is_some()
    }
}
```

### Task 5: Implement DoH-aware WebSocket connect

**Files:**
- Create: `cli/src/doh/ws.rs`

- [ ] **Step 1: Write `ws.rs`**

```rust
use anyhow::{Context, Result};
use std::sync::Arc;
use tokio::net::TcpStream;
use tokio_rustls::TlsConnector;
use tokio_tungstenite::{connect_async, tungstenite::Message};

use super::manager::DohManager;

/// WebSocket stream type when using DoH proxy (TLS over TCP).
pub type DohWsStream =
    tokio_tungstenite::WebSocketStream<tokio_rustls::client::TlsStream<TcpStream>>;

/// WebSocket stream type for standard connection (same as current daemon.rs).
pub type StdWsStream =
    tokio_tungstenite::WebSocketStream<tokio_tungstenite::MaybeTlsStream<TcpStream>>;

/// Connect to a WebSocket URL, using DoH proxy if available.
///
/// When DohManager has a proxy node:
///   1. TCP connect to node.ip:443
///   2. TLS handshake with SNI = node.host
///   3. WS handshake over the TLS stream
///
/// When no proxy (direct mode or no cache):
///   Standard connect_async(url)
pub async fn doh_connect_ws(
    url: &str,
    doh: &DohManager,
) -> Result<DohWsConnection> {
    if let Some((host, addr)) = doh.resolve_override() {
        // Proxy mode: manual TCP → TLS → WS
        let tcp = TcpStream::connect(addr)
            .await
            .context("DoH proxy TCP connect failed")?;

        let tls_config = rustls::ClientConfig::builder()
            .with_root_certificates(root_certs())
            .with_no_client_auth();

        let connector = TlsConnector::from(Arc::new(tls_config));
        let server_name = rustls::pki_types::ServerName::try_from(host.as_str())
            .map_err(|e| anyhow::anyhow!("invalid SNI: {}", e))?
            .to_owned();

        let tls_stream = connector
            .connect(server_name, tcp)
            .await
            .context("DoH proxy TLS handshake failed")?;

        let (ws, resp) = tokio_tungstenite::client_async(url, tls_stream)
            .await
            .context("DoH proxy WS handshake failed")?;

        Ok(DohWsConnection::Proxy(ws))
    } else {
        // Direct mode: standard connect
        let (ws, _) = connect_async(url)
            .await
            .context("WS connect failed")?;
        Ok(DohWsConnection::Direct(ws))
    }
}

fn root_certs() -> rustls::RootCertStore {
    let mut store = rustls::RootCertStore::empty();
    store.extend(webpki_roots::TLS_SERVER_ROOTS.iter().cloned());
    store
}

/// Enum to hold either connection type, both implement the same WS operations.
pub enum DohWsConnection {
    Direct(StdWsStream),
    Proxy(DohWsStream),
}

impl DohWsConnection {
    /// Send a message.
    pub async fn send(&mut self, msg: Message) -> Result<()> {
        use futures_util::SinkExt;
        match self {
            Self::Direct(ws) => ws.send(msg).await.map_err(Into::into),
            Self::Proxy(ws) => ws.send(msg).await.map_err(Into::into),
        }
    }

    /// Receive the next message.
    pub async fn next(&mut self) -> Option<Result<Message, tokio_tungstenite::tungstenite::Error>> {
        use futures_util::StreamExt;
        match self {
            Self::Direct(ws) => ws.next().await,
            Self::Proxy(ws) => ws.next().await,
        }
    }
}
```
